\section{\textsc{Salatsauce mit Joghurt}}

\subsection*{Zutaten für 300ml:}

\begin{tabular}{p{7.5cm} p{7.5cm}}
	& \\
	250g Joghurt & 10ml Orangensaft \\
	10ml Öl & 5ml Zitronensaft \\
	Worcestersauce & Salz \& Pfeffer
\end{tabular}

	\\
